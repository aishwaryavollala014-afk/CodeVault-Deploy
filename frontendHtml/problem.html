Given a non-negative integer represented as a linked list of digits, add one to the integer. The digits are stored such that the most significant digit is at the head.

Example

Input: 1 -> 2 -> 3 Output: 1 -> 2 -> 4

Constraints

The number of nodes is in the range [1, 100]. Each node's value is in [0, 9].

class Solution:
    def plusOne(self, head: ListNode) -> ListNode:
        def dfs(node):
            if not node:
                return 1
            carry = dfs(node.next)
            total = node.val + carry
            node.val = total % 10
            return total // 10

        carry = dfs(head)
        if carry:
            new = ListNode(1)
            new.next = head
            return new
        return head
